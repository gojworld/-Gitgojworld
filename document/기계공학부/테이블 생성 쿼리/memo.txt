CREATE TABLE `memo` (
	`MEMO_SEQ` INT(50) NOT NULL AUTO_INCREMENT COMMENT '메모 고유 번호',
	`UPR_MEMO_NB` INT(50) NULL DEFAULT '0' COMMENT '상위 메모 번호',
	`WRIR_M_SEQ` INT(11) NULL DEFAULT NULL COMMENT '사용자순번',
	`MENU_CD` VARCHAR(50) NULL DEFAULT NULL COMMENT '메뉴 코드',
	`WRIR_NM` VARCHAR(50) NULL DEFAULT NULL COMMENT '작성자 명',
	`CONT` TEXT NULL COMMENT '메모 내용',
	`PWD` TEXT NULL COMMENT '비밀번호',
	`BOARD_SEQ` INT(50) NULL DEFAULT NULL COMMENT '게시판 순번',
	`INPUT_DM` VARCHAR(19) NULL DEFAULT NULL COMMENT '등록 일자',
	PRIMARY KEY (`MEMO_SEQ`),
	INDEX `BS_INDEX` (`BOARD_SEQ`, `MENU_CD`, `WRIR_M_SEQ`)
)
COMMENT='한줄메모\r\n'
COLLATE='utf8_general_ci'
ENGINE=MyISAM
AUTO_INCREMENT=1004254;
