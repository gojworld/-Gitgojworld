CREATE TABLE `zipcode` (
	`SEQ` INT(5) UNSIGNED NOT NULL COMMENT '고유값',
	`ZIPCODE` CHAR(7) NOT NULL COMMENT '우편번호',
	`SIDO` VARCHAR(4) NULL DEFAULT NULL COMMENT '시 도',
	`GUGUN` VARCHAR(17) NULL DEFAULT NULL COMMENT '구 군',
	`DONG` VARCHAR(52) NULL DEFAULT NULL COMMENT '동',
	`BUNJI` VARCHAR(17) NULL DEFAULT NULL COMMENT '번지',
	PRIMARY KEY (`ZIPCODE`, `SEQ`)
)
COMMENT='우편번호'
COLLATE='utf8_general_ci'
ENGINE=MyISAM;
