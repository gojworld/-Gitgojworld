CREATE TABLE `board` (
	`BOARD_SEQ` INT(11) NOT NULL AUTO_INCREMENT COMMENT '게시판 순번',
	`LANG_CD` VARCHAR(30) NULL DEFAULT NULL COMMENT '언어 코드',
	`MENU_CD` VARCHAR(30) NULL DEFAULT NULL COMMENT '메뉴 코드(분류)',
	`CAT_CD` VARCHAR(30) NULL DEFAULT NULL COMMENT '카테고리 코드',
	`TTL` VARCHAR(200) NULL DEFAULT NULL COMMENT '제목',
	`CONT` MEDIUMTEXT NULL COMMENT '내용',
	`INQ_CNT` INT(11) NULL DEFAULT '0' COMMENT '조회 수',
	`WRIR_NM` VARCHAR(50) NULL DEFAULT NULL COMMENT '작성자 명',
	`UF_SEQ` INT(11) NULL DEFAULT '0' COMMENT '파일순번',
	`WRIR_M_SEQ` INT(11) NULL DEFAULT NULL COMMENT '사용자순번',
	`ANNC_YN` VARCHAR(11) NULL DEFAULT NULL COMMENT '공지 여부',
	`IP_ADDR` VARCHAR(50) NULL DEFAULT NULL COMMENT '아이피 주소',
	`PWD` VARCHAR(50) NULL DEFAULT NULL COMMENT '비밀번호',
	`INPUT_DM` VARCHAR(19) NULL DEFAULT NULL COMMENT '등록 일자',
	PRIMARY KEY (`BOARD_SEQ`),
	INDEX `BOARD_INDEX` (`LANG_CD`, `MENU_CD`, `CAT_CD`, `WRIR_M_SEQ`)
)
COMMENT='게시판'
COLLATE='utf8_general_ci'
ENGINE=MyISAM
AUTO_INCREMENT=1009633;
