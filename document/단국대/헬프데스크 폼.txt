<script type="text/javascript">
	$( document ).ready( function () {
		$("#wrap").css({"padding-top":"0px"});
		$("#header").hide();
		$( "#notcCont" ).val("핸드폰 기종 :\r\n프로그램명 :\r\n내용(자세히 적어주세요) :\r\n");
		$( "#saveBtn" ).click( function ( e ) {
			e.preventDefault();
			if ( $( "#notcTtl" ).val().trim() == null || $( "#notcTtl" ).val().trim() == "" ) {
				$( "#notcTtl" ).val( "" );
				alert( "제목은 필수입력값입니다." );
				$( "#notcTtl" ).focus();
				return false;
			}
			if ( $( "#notcCont" ).val().trim() == null || $( "#notcCont" ).val().trim() == "" ) {
				$( "#notcCont" ).val( "" );
				alert( "내용을 입력해주십시오." );
				$( "#notcCont" ).focus();
				return false;
			}
			if ( !confirm( "저장 하시겠습니까?" ) ) {
				return false;
			} else {
				e.preventDefault();
				$.ajax( {
					"url" : "comm/syst/notc/saveBoardBasMo.do" ,
					"data" : $( "#notc" ).serialize() ,
					"type" : "post" ,
					"dataType" : "json" ,
					"success" : function ( result ) {
						alert( "저장되었습니다." );
						location.reload();
					}
				} )
			}
			e.preventDefault();
		} );
	} );
</script>
