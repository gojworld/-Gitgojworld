<%@ page language="java" contentType="text/html; charset=utf-8" pageEncoding="utf-8"%>
<%@ include file="/WEB-INF/includes/taglibs.jsp"%>
<script type="text/javascript">
	var _self={
		"page":1,
		"bbsId":"1096",
		"category":"죽전",
		"pageSync":true,
		'template':'<li><a><strong class="cdt {color}">{category}</strong><strong class="fc_gray">{createDate} [{dept}]</strong><br /><strong>제목</strong></a><li>',
		"list":function(){
			$.ajax({
				type:"get",
				url:"etcs/ptal/findPtalAnncListToJSON.do",
				async:false,
				dataType:"json",
				data:{
					"page":_self.page
					,"bbsId":_self.bbsId
					,"category":_self.category
				},
				success:function(data){
					var $listUl=$("#listUl");
					if(data.ptalAnncJList.length!=20){
						_self.pageSync=false;
					}
					var category={
						"공통":"gray",
						"죽전":"blue",
						"천안":"green"
					}
					$.each(data.ptalAnncJList,function(index,ptalAnnc){
						$listUl.addRow(_self.template,{
							"createDate":ptalAnnc.createDate,
							"dept":ptalAnnc.dept,
							"category":ptalAnnc.category,
							"color":category[ptalAnnc.category],
							"bbsId":ptalAnnc.bbsId
						});
					});
				}
			});
		}
	};
	$(document).ready(function(){
		_self.list();
		$(function(){
			$(window).scroll(function(){
				if($(window).scrollTop()>=$(document).height()-$(window).height()){
					if(_self.pageSync){
						_self.page++;
						_self.list();
					}
				}
			});
		});
	});
</script>
<div class="notice_wrap">
	<ul class="list_notice" id="listUl">
	</ul>
</div>
