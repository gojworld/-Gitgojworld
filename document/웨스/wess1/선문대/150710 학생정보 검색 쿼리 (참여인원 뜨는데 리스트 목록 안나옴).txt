	SELECT  	A.JIM_SEQ,
					A.YEAR,
					A.SEASON,
					B.NAME,
			        B.STUDENT_NUM,
			        B.STD_NUM,
			        CONCAT(B.DEPT, ' ', B.MAJOR) AS MAJOR,
			        B.GRADE,
			        B.SEX,
					B.SEX AS SEX_TXT,
			        B.HP1 AS HP,
			        B.HP1, B.HP2, B.HP3,
			        CONCAT(B.EMAIL1, '@', B.EMAIL2) AS EMAIL,
			        B.TOT_CREDIT,
			        B.MARK,
			        B.BL_YN,
			        B.DURING_STATUS AS DURING_STATUS,
					getCodeName(B.DURING_STATUS) DURING_STATUS_TXT,
			        A.DEGREE ,
			        D.COMP_NAME,
			        (SELECT STATUS FROM JOB_TOTAL_REPORT WHERE JIM_SEQ = A.JIM_SEQ) REPORT_STATUS,
					IFNULL(getCodeName((SELECT STATUS FROM JOB_TOTAL_REPORT WHERE JIM_SEQ = A.JIM_SEQ)), '작성전') REPORT_STATUS_TXT,
					A.STATUS,
					getCodeName(A.STATUS) STATUS_TXT,
					CASE
						IFNULL(A.COMP_STATUS, '')
						WHEN '' THEN '심사전'
						WHEN NULL THEN '심사전'
					ELSE getCodeName(A.COMP_STATUS)
					END COMP_STATUS_TXT,
					A.COMP_STATUS,
					CONCAT(C.INTERN_S_DATE, '~' , C.INTERN_E_DATE) DATE,
					CAST(CONCAT(E.TITLE, '(', getCodeName(E.INTERN_DIV), ',' , E.S_YEAR, '.',  E.S_MONTH, '.',  E.S_DAY, '~', E.E_YEAR, '.', E.E_MONTH, '.', E.E_DAY, ')') AS CHAR) TITLE,
					CAST(CONCAT(E.TITLE, '(', getCodeName(E.INTERN_DIV), ') - ', D.COMP_NAME, ' : ',  C.INTERN_S_DATE, '~',  C.INTERN_E_DATE) AS CHAR) POP_TITLE,
					C.INTERN_S_DATE,
					C.INTERN_E_DATE,
					C.JID_SEQ,
					getCodeName(A.SCH_STATUS) SCH_STATUS_TXT,
					getCodeName(A.EVAL_STATUS) EVAL_STATUS_TXT,					
					A.SCH_STATUS,
					A.EVAL_STATUS,
					E.JS_SEQ,
					E.TITLE S_TITLE,
					C.DEPT,
					A.PROF_VALUATION,
					IFNULL((SELECT SUM(CASE POINT WHEN '매우우수' THEN 10 WHEN '우수' THEN 8 WHEN '보통' THEN 6 WHEN '미흡' THEN 4 WHEN '부족' THEN 2 END)
						 FROM JOB_INTERN_VALUATION WHERE JIM_SEQ = A.JIM_SEQ), 0) COMP_VALUATION,
                    (SELECT COUNT(0) FROM JOB_INTERN_WORKDAY WHERE JIM_SEQ = A.JIM_SEQ) WORK_CNT,
                    CASE WHEN (SELECT STUDENT_NUM FROM JOB_STUdENT WHERE STUDENT_NUM = D.REG_ID) IS NULL THEN 'N' ELSE 'Y' END AS STD_REQ_YN 
		FROM    	JOB_INTERN_MANAGE A
			        INNER JOIN JOB_STUDENT B ON A.STUDENT_NUM = B.STUDENT_NUM
			        INNER JOIN JOB_INTERN_DETAIL C ON A.JID_SEQ = C.JID_SEQ
			        INNER JOIN JOB_COMP D ON C.COMP_CODE = D.COMP_CODE
			        INNER JOIN JOB_SCHEDULE E
			        	ON C.JS_SEQ = E.JS_SEQ AND E.DEL_YN = 'N'
		WHERE A.DEL_YN = 'N'
AND B.NAME LIKE '조덕진'
